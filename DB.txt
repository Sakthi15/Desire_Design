create table PAYMENT
(
Payment_id int identity(1,1),
payment_code AS 'DDPAY' + RIGHT('00'+ CONVERT(VARCHAR(255),Payment_id),255) persisted ,
Aadhar_number varchar(12) foreign key references ONLINE_CONSULTATION(Aadhar_number) not null,
Item_code varchar(20),
Amount varchar(10) not null,
Card_Holder varchar(255) not null,
Card_Number varchar(12) not null,
Expiry_Datee varchar(10) not null,
CVV varchar(3) not null,
)

create table FEEDBACK
(

Rating varchar(2) not null

create procedure online_insert(
@Namee varchar(255) ,
@Aadhar_number varchar(12),
@Mobile varchar(10) ,
@Emailid varchar(30) ,
@Address varchar(255) ,
@Property_Name varchar(255))
as
begin
insert into ONLINE_CONSULTATION values(@Namee,@Aadhar_number,@Mobile,@Emailid,@Address,@Property_Name)
end

select * from ONLINE_CONSULTATION


create procedure professional_insert(
@Namee varchar(255),
@Mobile varchar(10) ,
@Emailid varchar(30),
@Address varchar(255),
@Internship_name varchar(255) )
as
begin
declare @co int
set @co=(select count(*) from PROFESSIONAL_REGSISTRATION where Emailid=@Emailid)
if @co<1
begin
insert into PROFESSIONAL_REGSISTRATION values(@Namee,@Mobile,@Emailid,@Address,@Internship_name)
end
end


create procedure feedback_insert(
@Aadhar_number varchar(12),
@Comment varchar(255) ,
@Rating varchar(2))
as
begin
insert into FEEDBACK values(@Aadhar_number,@Comment,@Rating)
end

create procedure payment_insert(
@Aadhar_number varchar(12),
@Amount varchar(10) ,
@Item_code varchar(20),
@Card_Holder varchar(255) ,
@Card_Number varchar(12) ,
@Expiry_Datee varchar(10),
@CVV varchar(3))
as
begin
insert into PAYMENT values(@Aadhar_number,@Amount,@Item_code,@Card_Holder,@Card_Number,@Expiry_Datee,@CVV)
end
